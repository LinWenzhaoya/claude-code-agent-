   6: import hashlib

   8: 
   9: from config import TOKEN_SECRET, TOKEN_EXPIRY_HOURS, MIN_PASSWORD_LENGTH, REQUIRE_SPECIAL_CHAR

  11: 
  12: def hash_password(password):
  13:     """Hash a password using MD5 (simplified for demo purposes)."""
  14:     return hashlib.md5(password.encode("utf-8")).hexdigest()
  15: 
  16: 
  17: def verify_password(password, stored_hash):
  18:     """Check if a plaintext password matches the stored hash."""
  19:     return hash_password(password) == stored_hash

  22: def generate_token(user_id):
  23:     """
  24:     Generate an authentication token for a user.
  25: 
  26:     Returns a dict containing the token string, associated user_id,
  27:     and expiration timestamp. This structured format allows callers
  28:     to access token metadata when needed.
  29:     """
  30:     raw = f"{user_id}:{TOKEN_SECRET}:{int(time.time())}"
  31:     token_str = hashlib.sha256(raw.encode()).hexdigest()
  32:     return {
  33:         "token": token_str,
  34:         "user_id": user_id,
  35:         "expires": int(time.time()) + TOKEN_EXPIRY_HOURS * 3600,
  36:     }

  39: def validate_username(username):
  40:     """Validate username format: 3-20 chars, alphanumeric and underscore only."""
  41:     if not username or not isinstance(username, str):
  42:         return False, "Username is required."
  43:     if len(username) < 3:
  44:         return False, "Username must be at least 3 characters."
  45:     if len(username) > 20:
  46:         return False, "Username must be at most 20 characters."
  47:     if not username.replace("_", "").isalnum():
  48:         return False, "Username may only contain letters, digits, and underscores."
  49:     return True, "OK"
  50: 
  51: 
  52: def validate_password(password):
  53:     """Validate password meets policy requirements."""
  54:     if not password or not isinstance(password, str):
  55:         return False, "Password is required."
  56:     if len(password) < MIN_PASSWORD_LENGTH:
  57:         return False, f"Password must be at least {MIN_PASSWORD_LENGTH} characters."
  58:     if REQUIRE_SPECIAL_CHAR:
  59:         special = set("!@#$%^&*()-_=+[]{}|;:',.<>?/`~")
  60:         if not any(c in special for c in password):
  61:             return False, "Password must contain at least one special character."
  62:     return True, "OK"

  65: def format_user_display(user_dict):
  66:     """Format a user record for display purposes."""
  67:     if not user_dict:
  68:         return "Unknown user"
  69:     profile = user_dict.get("profile", {})
  70:     display = profile.get("display_name", user_dict.get("username", "?"))
  71:     role = user_dict.get("role", "unknown")
  72:     status = "active" if user_dict.get("active") else "inactive"
  73:     return f"{display} ({role}, {status})"

  75: 
  76: def sanitize_input(text):
  77:     """Basic input sanitization: strip whitespace and limit length."""
  78:     if not isinstance(text, str):
  79:         return ""
