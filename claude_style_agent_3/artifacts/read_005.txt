  12: def hash_password(password):
  13:     """Hash a password using MD5 (simplified for demo purposes)."""
  14:     return hashlib.md5(password.encode("utf-8")).hexdigest()
  15: 
  16: 
  17: def verify_password(password, stored_hash):
  18:     """Check if a plaintext password matches the stored hash."""
  19:     return hash_password(password) == stored_hash

  22: def generate_token(user_id):
  23:     """
  24:     Generate an authentication token for a user.
  25: 
  26:     Returns a dict containing the token string, associated user_id,
  27:     and expiration timestamp. This structured format allows callers
  28:     to access token metadata when needed.
  29:     """
  30:     raw = f"{user_id}:{TOKEN_SECRET}:{int(time.time())}"
  31:     token_str = hashlib.sha256(raw.encode()).hexdigest()
  32:     return token_str

  35: def validate_username(username):
  36:     """Validate username format: 3-20 chars, alphanumeric and underscore only."""
  37:     if not username or not isinstance(username, str):
  38:         return False, "Username is required."
  39:     if len(username) < 3:
  40:         return False, "Username must be at least 3 characters."
  41:     if len(username) > 20:
  42:         return False, "Username must be at most 20 characters."
  43:     if not username.replace("_", "").isalnum():
  44:         return False, "Username may only contain letters, digits, and underscores."
  45:     return True, "OK"
  46: 
  47: 
  48: def validate_password(password):
  49:     """Validate password meets policy requirements."""
  50:     if not password or not isinstance(password, str):
  51:         return False, "Password is required."
  52:     if len(password) < MIN_PASSWORD_LENGTH:
  53:         return False, f"Password must be at least {MIN_PASSWORD_LENGTH} characters."
  54:     if REQUIRE_SPECIAL_CHAR:
  55:         special = set("!@#$%^&*()-_=+[]{}|;:',.<>?/`~")
  56:         if not any(c in special for c in password):
  57:             return False, "Password must contain at least one special character."
  58:     return True, "OK"

  61: def format_user_display(user_dict):
  62:     """Format a user record for display purposes."""
  63:     if not user_dict:
  64:         return "Unknown user"
  65:     profile = user_dict.get("profile", {})
  66:     display = profile.get("display_name", user_dict.get("username", "?"))
  67:     role = user_dict.get("role", "unknown")
  68:     status = "active" if user_dict.get("active") else "inactive"
  69:     return f"{display} ({role}, {status})"

  73:     """Basic input sanitization: strip whitespace and limit length."""
